1.2 改修内容

L112 そのまま（パスは変えない）。L115-142 を以下に置換:

# 親 Plan の §3「magi-routing.json 正式スキーマ」準拠。
# CEO + ちゃっぴーレビュー 2026-05-13 確定。
MAGI_ROUTING_SCHEMA_VERSION = 2  # v1 = current/switched_at/switched_by のみ、v2 = previous/mode/reason/lock 追加

_VALID_CTO = ("claude", "codex")
_VALID_MODE = ("normal", "acting_cto", "switching")

def _now_jst_iso() -> str:
    """JST ISO 8601。親 Plan §3 が JST `+09:00` を要件化。"""
    from datetime import timezone, timedelta
    return datetime.now(timezone(timedelta(hours=9))).strftime("%Y-%m-%dT%H:%M:%S+09:00")

def _read_magi_routing() -> dict:
    """magi-routing.json を読んで v2 スキーマで返す。
    v1 ファイルが残っていれば v2 にマイグレーションした上で返す（migrate-on-read）。
    """
    if not MAGI_ROUTING_FILE.exists():
        default = {
            "current": "claude",
            "previous": None,
            "mode": "normal",
            "switched_at": _now_jst_iso(),
            "switched_by": "default",
            "reason": "init",
            "lock": False,
        }
        _atomic_write_magi_routing(default)
        return default
    try:
        data = json.loads(MAGI_ROUTING_FILE.read_text())
    except Exception:
        return {
            "current": "claude",
            "previous": None,
            "mode": "normal",
            "switched_at": "",
            "switched_by": "error",
            "reason": "read-error",
            "lock": False,
        }
    # v1 → v2 マイグレーション（フィールドが無ければ補う）
    data.setdefault("previous", None)
    data.setdefault("mode", "normal" if data.get("current") == "claude" else "acting_cto")
    data.setdefault("reason", "legacy")
    data.setdefault("lock", False)
    return data

def _atomic_write_magi_routing(data: dict) -> None:
    """tmp ファイル → os.replace で atomic に差し替える（途中で読まれても壊れた JSON を見せない）。
    親 Plan §3 必須要件。
    """
    MAGI_ROUTING_FILE.parent.mkdir(parents=True, exist_ok=True)
    tmp = MAGI_ROUTING_FILE.with_suffix(".json.tmp")
    tmp.write_text(json.dumps(data, ensure_ascii=False, indent=2))
    tmp.chmod(0o644)
    os.replace(tmp, MAGI_ROUTING_FILE)  # POSIX atomic rename(2)

def _write_magi_routing(
    current: str,
    switched_by: str,
    *,
    mode: str = "normal",
    reason: str = "manual",
    lock: bool = False,
    previous: str | None = None,
) -> dict:
    """magi-routing.json を atomic update する。
    引数:
      current      "claude" | "codex"
      switched_by  "CEO" | "watchdog" | "auto-7pct" | "rollback" 等
      mode         "normal" | "acting_cto" | "switching"
      reason       "manual" | "usage-70pct" | "usage-limit-hit" | "switch-failure" | "rollback" | "init" | "legacy"
      lock         切替処理中の排他フラグ
      previous     None なら現在の current を読んで自動セット
    戻り値: 書き込んだ data dict
    """
    assert current in _VALID_CTO, f"invalid current: {current}"
    assert mode in _VALID_MODE, f"invalid mode: {mode}"
    prev_data = _read_magi_routing()
    if previous is None:
        previous = prev_data.get("current")
    data = {
        "current": current,
        "previous": previous,
        "mode": mode,
        "switched_at": _now_jst_iso(),
        "switched_by": switched_by,
        "reason": reason,
        "lock": lock,
    }
    _atomic_write_magi_routing(data)
    return data

def _set_routing_lock(lock: bool) -> dict:
    """lock フィールドだけ更新（atomic）。switching 処理の入口で True、終了 / rollback で False。
    他のフィールドは現状維持。
    """
    data = _read_magi_routing()
    data["lock"] = lock
    _atomic_write_magi_routing(data)
    return data

def _current_cto() -> str:
    """現在の CTO（'claude' or 'codex'）を routing から取得。読み失敗時は 'claude' fallback。"""
    return _read_magi_routing().get("current", "claude")

def _current_cto_tmux() -> str:
    """現在の CTO の tmux セッション名。"""
    return MAGI_TMUX_SESSION[_current_cto()]

def _is_tmux_alive(session_name: str) -> bool:
    """指定 tmux session が has-session で通るか。"""
    try:
        result = subprocess.run(
            ["tmux", "has-session", "-t", session_name],
            capture_output=True, text=True, timeout=5,
        )
        return result.returncode == 0
    except Exception:
        return False

2.2.4 is_codechan_idle() 改修（L2931-2952）

current=codex の場合は 常に True（codex の todos JSON を bridge は持っていない、idle 判定の意味がないため）。current=claude のみ既存ロジック。

def is_codechan_idle() -> bool:
    """現 CTO がアクティブタスクを持っていないか。
    routing=codex の時は判定手段がない → True（idle 扱い）。
    """
    if _current_cto() != "claude":
        return True
    try:
        import json as _json
        session_id = get_session_id()
        if not session_id:
            return True
        todos_dir = Path.home() / ".claude" / "todos"
        if not todos_dir.exists():
            return True
        for f in todos_dir.glob("*.json"):
            if session_id in f.name:
                tasks = _json.loads(f.read_text())
                for task in tasks:
                    if task.get("status") == "in_progress":
                        return False
                return True
        return True
    except Exception:
        return True

def restart_codechan():

4.3 新規同期関数 perform_cto_switch()（telegram_bot.py、restart_codechan 直後 = 元 L3102 付近に追加）

def perform_cto_switch(
    target: str,
    switched_by: str,
    reason: str,
) -> dict:
    """CTO 切替の本体。tmux kill + tmux 起動 + routing 更新 + 監査ログ + 失敗時ロールバック。
    引数:
      target       "claude" | "codex"（切替先）
      switched_by  "CEO" | "watchdog" | "auto-7pct" 等
      reason       "manual" | "usage-70pct" 等
    戻り値: {
      "ok": bool,
      "from": str,            # 切替元
      "to": str,              # 切替先
      "stages": list[dict],   # 各 stage の {name, ok, detail}
      "final_routing": dict,  # 完了後 / rollback 後の routing 状態
      "rolled_back": bool,
      "reason": str,
    }
    """
    assert target in _VALID_CTO
    current = _read_magi_routing()
    from_cto = current.get("current", "claude")
    if current.get("lock"):
        return {
            "ok": False, "from": from_cto, "to": target,
            "stages": [], "final_routing": current, "rolled_back": False,
            "reason": "already-locked",
        }
    if from_cto == target:
        return {
            "ok": True, "from": from_cto, "to": target,
            "stages": [{"name": "noop", "ok": True, "detail": "既に target"}],
            "final_routing": current, "rolled_back": False, "reason": "noop",
        }

    stages: list[dict] = []
    # Stage 0: lock + mode=switching
    _write_magi_routing(
        current=from_cto, switched_by=switched_by,
        mode="switching", reason=reason, lock=True, previous=from_cto,
    )
    _audit_log({
        "actor": switched_by, "mode": "switching", "event": "switch",
        "routing_before": from_cto, "routing_after": target,
        "command": _command_for_switched_by(switched_by, target),
        "summary": f"switch start: {from_cto} → {target}", "result": "start",
        "reason": reason,
    })

    try:
        # Stage 1: 切替元 tmux kill
        kill_target = MAGI_TMUX_SESSION[from_cto]
        kill_rc = subprocess.run(
            ["tmux", "kill-session", "-t", kill_target],
            capture_output=True, timeout=10,
        ).returncode
        stages.append({"name": "kill-source", "ok": True, "detail": f"{kill_target} rc={kill_rc}"})
        # kill_rc != 0（既に死んでた）も成功扱い（=想定範囲、目的は tmux 不在）

        # Stage 2: routing を target に更新（mode=switching 維持、lock=True 維持）
        _write_magi_routing(
            current=target, switched_by=switched_by,
            mode="switching", reason=reason, lock=True, previous=from_cto,
        )
        stages.append({"name": "routing-flip", "ok": True, "detail": f"current={target}"})

        # Stage 3: 切替先 tmux 起動
        start_result = start_codechan()  # routing.current=target に従って _start_*_cto() が走る
        if start_result not in ("new", "already_running"):
            stages.append({"name": "start-target", "ok": False, "detail": f"result={start_result}"})
            raise RuntimeError(f"start_codechan returned {start_result}")
        stages.append({"name": "start-target", "ok": True, "detail": f"result={start_result}"})

        # Stage 4: mode 確定 + lock 解除
        final_mode = "acting_cto" if target == "codex" else "normal"
        final = _write_magi_routing(
            current=target, switched_by=switched_by,
            mode=final_mode, reason=reason, lock=False, previous=from_cto,
        )
        stages.append({"name": "finalize", "ok": True, "detail": f"mode={final_mode}"})

        _audit_log({
            "actor": switched_by, "mode": final_mode, "event": "switch",
            "routing_before": from_cto, "routing_after": target,
            "command": _command_for_switched_by(switched_by, target),
            "summary": f"switch success: {from_cto} → {target}", "result": "success",
            "reason": reason,
        })
        return {
            "ok": True, "from": from_cto, "to": target,
            "stages": stages, "final_routing": final, "rolled_back": False, "reason": reason,
        }

    except Exception as e:
        # ロールバック（§7）
        rollback_result = _rollback_to_claude(
            failed_target=target, failed_reason=str(e), switched_by=switched_by,
        )
        stages.append({"name": "rollback", "ok": rollback_result["ok"], "detail": str(e)})
        return {
            "ok": False, "from": from_cto, "to": target,
            "stages": stages, "final_routing": rollback_result["final_routing"],
            "rolled_back": True, "reason": f"switch-failure: {e}",
        }


def _command_for_switched_by(switched_by: str, target: str) -> str | None:
    if switched_by == "CEO":
        return "!decchan" if target == "codex" else "!codechan"
    return None


def _format_switch_result(result: dict) -> str:
    """perform_cto_switch() の戻り値を Discord 表示用テキストに整形。"""
    if result["ok"]:
        if result["to"] == "codex":
            return (
                f"🔄 代打 CTO モード ON。でっくすちゃんが CTO に交代。\n"
                f"#一般 全担当します（mention 不要）。コードちゃん休憩中。"
            )
        return f"✅ コードちゃん復帰。でっくすちゃん待機。"
    # 失敗
    stage_lines = "\n".join([f"  - {s['name']}: {'OK' if s['ok'] else 'NG'} ({s['detail']})" for s in result["stages"]])
    if result["rolled_back"]:
        return (
            f"⚠️ 切替失敗・Claude へ復帰しました\n"
            f"理由: {result['reason']}\n"
            f"stages:\n{stage_lines}"
        )
    return f"❌ 切替失敗（ロールバック未実行）\n理由: {result['reason']}\nstages:\n{stage_lines}"

■ 詳細設計 §7: ロールバック手順

7.1 親 Plan §1 の要件

def _rollback_to_claude(
    *,
    failed_target: str,
    failed_reason: str,
    switched_by: str,
) -> dict:
    """切替失敗時の巻き戻し。常に claude CTO に戻す（親 Plan §1 ロールバック手順準拠）。
    戻り値: {"ok": bool, "final_routing": dict, "stages": list[dict]}
    """
    stages: list[dict] = []
    _audit_log({
        "actor": switched_by, "mode": "switching", "event": "rollback",
        "routing_before": failed_target, "routing_after": "claude",
        "command": None,
        "summary": f"rollback start (failed switch to {failed_target})", "result": "start",
        "reason": failed_reason,
    })
    try:
        # Stage R1: codex tmux を kill（あれば）
        kill_rc = subprocess.run(
            ["tmux", "kill-session", "-t", "decchan"], capture_output=True, timeout=10,
        ).returncode
        stages.append({"name": "rollback-kill-decchan", "ok": True, "detail": f"rc={kill_rc}"})

        # Stage R2: routing を claude / mode=normal / lock=false に
        _write_magi_routing(
            current="claude", switched_by="rollback",
            mode="normal", reason="switch-failure", lock=False, previous=failed_target,
        )
        stages.append({"name": "rollback-routing", "ok": True, "detail": "current=claude"})

        # Stage R3: claude tmux を起動（routing.current=claude なので _start_claude_cto() が走る）
        start_result = start_codechan()
        ok = start_result in ("new", "already_running")
        stages.append({"name": "rollback-start-claude", "ok": ok, "detail": f"result={start_result}"})

        final = _read_magi_routing()
        _audit_log({
            "actor": switched_by, "mode": "normal", "event": "rollback",
            "routing_before": failed_target, "routing_after": "claude",
            "command": None,
            "summary": f"rollback {'success' if ok else 'partial'}", "result": "success" if ok else "partial",
            "reason": failed_reason,
        })
        return {"ok": ok, "final_routing": final, "stages": stages}

    except Exception as e:
        _audit_log({
            "actor": switched_by, "mode": "switching", "event": "rollback",
            "routing_before": failed_target, "routing_after": "claude",
            "command": None,
            "summary": f"rollback failed: {e}", "result": "failure",
            "reason": failed_reason,
        })
        return {"ok": False, "final_routing": _read_magi_routing(), "stages": stages}

7.3 ロールバック対象範囲

  • 「戻れ（!codechan）」の切替が失敗した場合も同じく claude に巻き戻す（target=claude が失敗 → claude をもう一度起動し直す試行 = 同じ動作なので冪等）
  • ロールバック自体が失敗した場合 = Discord に ❌ ロールバックも失敗 を出して処理は停止（CEO 手動介入）。watchdog は mode!=switching なら自動再起動を試みる

■ 段階的実装順序（PR 分割案）

本 Plan の単一 PR で すべて一括実装は規模的にギリギリ（+400 LOC、レベル C 相当）。レビュー負荷を考えると 2 段階分割を推奨:

PR-α: スキーマ拡張 + 抽象化（無害化）

  • §1 スキーマ拡張 + atomic write
  • §2 抽象化（_current_cto* ヘルパ + is_codechan_* / start_codechan / restart_codechan / _notify_codechan の routing-aware 化、ただし routing=codex のパスは launcher 不在で 必ず None を返す = 無害）
  • §6 監査ログヘルパ（呼び出しはまだ）

この時点で挙動変化なし（routing=claude のままなので既存挙動と等価、新規ヘルパは未使用）。

PR-β: 切替動作実装

  • §3 run_codex_for_chat role 注入
  • §4 !decchan / !codechan / !magi-status を perform_cto_switch 化
  • §5 watchdog の routing-aware 化（mode=switching スキップ）
  • §6 監査ログの呼び出し配線

PR-β merge と AiMiraiLabs 側の roles/acting_cto.md + decchan-launcher.sh 配備が揃った時点で実運用 ON。

PR 分割しない場合
